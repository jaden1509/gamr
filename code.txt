<body>
  <div class="game-container">
    <h1>Score Game</h1>
    <button onclick="increaseScore()" class="btn">Increase Score</button>
    <div class="scoreboard">
      <h3>Scoreboard</h3>
      <div id="scores"></div>
    </div>
  </div>

  <script>
    function increaseScore() {
      if (typeof(Storage) !== "undefined") {
        if (localStorage.score) {
          localStorage.score = Number(localStorage.score) + 1;
        } else {
          localStorage.score = 1;
        }
        updateScoreboard();
      } else {
        alert("Sorry, your browser does not support web storage. Scores cannot be saved.");
      }
    }

    function updateScoreboard() {
      var scoresDiv = document.getElementById("scores");
      scoresDiv.innerHTML = "";
      var scoreCount = localStorage.score || 0;

      for (var i = 1; i <= scoreCount; i++) {
        var scoreItem = document.createElement("p");
        scoreItem.innerText = "Score " + i;
        scoresDiv.appendChild(scoreItem);
      }
    }

    updateScoreboard();
  </script>
